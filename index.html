        const wifiForm = document.getElementById('wifiForm');

        const messageBox = document.getElementById('messageBox');


        // --- Dynamic Display Logic ---


        // Wi-Fi Network "Other" input and MAC Address section

        document.querySelectorAll('input[name="wifiNetwork"]').forEach(radio => {

            radio.addEventListener('change', (event) => {

                const otherNetworkInput = document.querySelector('input[name="wifiNetworkOther"]');

                const macSection = document.getElementById('macAddressSection');

                

                // Toggle "Other" network input

                if (event.target.value === 'Other') {

                    otherNetworkInput.classList.remove('hidden');

                    otherNetworkInput.setAttribute('required', 'required');

                } else {

                    otherNetworkInput.classList.add('hidden');

                    otherNetworkInput.removeAttribute('required');

                    otherNetworkInput.value = '';

                }


                // Toggle MAC Address section for "Guest"

                if (event.target.value === 'Guest') {

                    macSection.style.display = 'block';

                    document.querySelectorAll('input[name="provideMac"]').forEach(macRadio => macRadio.setAttribute('required', 'required'));

                } else {

                    macSection.style.display = 'none';

                    document.querySelectorAll('input[name="provideMac"]').forEach(macRadio => {

                        macRadio.removeAttribute('required');

                        macRadio.checked = false; // Uncheck MAC preference

                    });
